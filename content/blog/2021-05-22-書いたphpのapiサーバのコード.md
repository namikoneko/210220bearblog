function make_list_one($row){
    $list["id"] = $row->id;
    $list["title"] = $row->title;
    $list["postid"] = $row->postid;
    $list["jid"] = $row->jid;
    $list["text"] = $row->text;
    return $list;
}

function make_list_many($rows){
    $i = 0;
    foreach($rows as $row){
       $list[$i]["id"] = $row->id;
       $list[$i]["postid"] = $row->postid;
       $list[$i]["title"] = $row->title;
       $list[$i]["hr"] = $row->hr;
       $list[$i]["show"] = $row->show;
       $list[$i]["text"] = $row->text;
        $i++;
    }
    return $list;
}

Flight::start();
